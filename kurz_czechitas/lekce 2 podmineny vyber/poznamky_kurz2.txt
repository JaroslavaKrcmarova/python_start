import pandas as pd
food_nutrient = pd.read_csv("food_nutrient.csv")
food_nutrient.head()

#cisteni dat - mazani nepotrebnzch sloupcu
food_nutrient = food_nutrient.drop(columns=['data_points', 'derivation_id'])
#unikatni hodnoty
food_nutrient['name'].unique()
food_nutrient['name'].nunique()
food_nutrient['name'].value_count() #pocty nejcastej vzsyktovanych unik hodnto
food_nutrient['name'].value_count().head(15) #top15 nejcastej vyskyt. hodnot

# filtrace dle hodnot
food_nutrient[food_nutrient['name'] == 'Magnesium, Mg'] #filtrace zaznamu ktere splnuji podminku

podminka =  food_nutrient['name'] == 'Magnesium, Mg' #2.zpusob podmineneho filtrovani, podminku ukladam do promenne
magnesium = food_nutrient[podminka]#ukladam do promenne 
magnesium.describe() 	#popisni statistika, zjistim prumerny obsah napr
magnesium['amount'].describe() 	#popisni statistika konkretniho sloupecku
magnesium[magnesium['amount']>100] #dalsi filtrace
